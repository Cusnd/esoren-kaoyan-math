\subsection{公式索引}

\begin{knowledgeBox}
\textbf{倒数型整体换元恒等式}

\[
u^2+\frac{1}{u^2}=\left(u+\frac{1}{u}\right)^2-2,\qquad u\ne 0.
\]

\begin{itemize}
  \item \textbf{使用条件：} 原式中出现 \(u+\frac{1}{u}\) 或可通过同除 \(u^k\) 构造该整体，且 \(u\ne 0\)。
  \item \textbf{典型题型：} 已知 \(f\left(u+\frac{1}{u}\right)\) 反求 \(f(x)\)，以及含倒数结构的整体换元。
  \item \textbf{易错点：} 不能把题设参数 \(u\) 与最终自变量 \(x\) 混用；若出现 \(u-\frac{1}{u}\)，还要额外关注分支和符号。
  \item \textbf{关联题目：} MATH1-CALC-0001。
\end{itemize}

\textbf{反函数复合恒等式}

设 \(f:D\to R\) 一一对应，反函数为 \(f^{-1}:R\to D\)，则
\[
f^{-1}(f(x))=x,\qquad x\in D,
\]
\[
f(f^{-1}(y))=y,\qquad y\in R.
\]

\begin{itemize}
  \item \textbf{使用条件：} 原函数必须在给定定义域上一一对应；第一个恒等式中 \(x\) 属于原函数定义域，第二个恒等式中 \(y\) 属于原函数值域。
  \item \textbf{典型题型：} 反函数求值、复合函数化简、由图像或定义域值域判断反函数。
  \item \textbf{易错点：} \(f^{-1}(x)\) 是反函数，不是 \(\frac{1}{f(x)}\)；若忽略变量范围，可能把无定义的复合式误写成恒等式。
  \item \textbf{关联知识点：} 第 1 讲“反函数的定义、存在条件与图像关系”。
\end{itemize}

\textbf{倒数代换中的根式化简}

当 \(x>0\) 时，
\[
\sqrt{1+\frac{1}{x^2}}
=\sqrt{\frac{x^2+1}{x^2}}
=\frac{\sqrt{1+x^2}}{x}.
\]

\begin{itemize}
  \item \textbf{使用条件：} 必须明确 \(x>0\)。一般情况下 \(\sqrt{x^2}=|x|\)，不能直接写成 \(x\)。
  \item \textbf{典型题型：} 将关系式中的 \(x\) 换为 \(\frac{1}{x}\) 后化简右端。
  \item \textbf{易错点：} 忘记绝对值导致符号错误。
  \item \textbf{关联题目：} MATH1-CALC-0002。
\end{itemize}

\textbf{反双曲正弦与双曲正弦互为反函数}

\[
\ln\left(x+\sqrt{x^2+1}\right)=y
\quad\Longleftrightarrow\quad
x=\frac12\left(e^y-e^{-y}\right).
\]

等价地，
\[
\operatorname{arsinh}x=\ln\left(x+\sqrt{x^2+1}\right),
\qquad
\sinh x=\frac{e^x-e^{-x}}{2}.
\]

\begin{itemize}
  \item \textbf{使用条件：} \(x+\sqrt{x^2+1}>0\)，原函数在 \(\mathbb{R}\) 上有定义并一一对应。
  \item \textbf{典型题型：} 求 \(\ln\left(x+\sqrt{x^2+1}\right)\) 的反函数，或识别反双曲正弦、双曲正弦结构。
  \item \textbf{易错点：} 直接平方硬解容易变繁；倒数有理化可直接得到 \(e^{-y}=\sqrt{x^2+1}-x\)。
  \item \textbf{关联题目：} MATH1-CALC-0003。
\end{itemize}

\textbf{复合函数定义域条件}

若 \(f\) 的定义域为 \(D_f\)，\(g\) 的定义域为 \(D_g\)，则
\[
D_{f\circ g}=\{x\in D_g\mid g(x)\in D_f\}.
\]

\begin{itemize}
  \item \textbf{使用条件：} 先保证内层 \(g(x)\) 有意义，再保证内层输出能作为外层 \(f\) 的输入。
  \item \textbf{典型题型：} 复合函数定义域、分段函数复合、已知复合结果反求中间函数。
  \item \textbf{易错点：} 把复合函数定义域误认为内层函数定义域，漏掉外层函数对内层输出的限制。
  \item \textbf{关联题目：} MATH1-CALC-0004，MATH1-CALC-0005。
\end{itemize}

\textbf{平方复合反求中间函数}

若 \(f(t)=t^2\)，且
\[
f[\varphi(x)]=G(x),
\]
则
\[
[\varphi(x)]^2=G(x).
\]
若进一步已知 \(\varphi(x)\ge 0\)，则
\[
\varphi(x)=\sqrt{G(x)}.
\]

\begin{itemize}
  \item \textbf{使用条件：} \(G(x)\ge 0\)，并且必须有分支条件才能唯一确定 \(\varphi(x)\)。
  \item \textbf{典型题型：} 已知平方型外层函数和复合结果，反求内层函数。
  \item \textbf{易错点：} 没有 \(\varphi(x)\ge 0\) 等条件时，不能只取正根。
  \item \textbf{关联题目：} MATH1-CALC-0004。
\end{itemize}
\end{knowledgeBox}
