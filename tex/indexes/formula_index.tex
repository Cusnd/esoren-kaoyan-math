\subsection{公式索引}

\begin{knowledgeBox}
\textbf{倒数型整体换元恒等式}

\[
u^2+\frac{1}{u^2}=\left(u+\frac{1}{u}\right)^2-2,\qquad u\ne 0.
\]

\begin{itemize}
  \item \textbf{使用条件：} 原式中出现 \(u+\frac{1}{u}\) 或可通过同除 \(u^k\) 构造该整体，且 \(u\ne 0\)。
  \item \textbf{典型题型：} 已知 \(f\left(u+\frac{1}{u}\right)\) 反求 \(f(x)\)，以及含倒数结构的整体换元。
  \item \textbf{易错点：} 不能把题设参数 \(u\) 与最终自变量 \(x\) 混用；若出现 \(u-\frac{1}{u}\)，还要额外关注分支和符号。
  \item \textbf{关联题目：} \problemRef{MATH1-CALC-0001}。
\end{itemize}

\textbf{反函数复合恒等式}

设 \(f:D\to R\) 一一对应，反函数为 \(f^{-1}:R\to D\)，则
\[
f^{-1}(f(x))=x,\qquad x\in D,
\]
\[
f(f^{-1}(y))=y,\qquad y\in R.
\]

\begin{itemize}
  \item \textbf{使用条件：} 原函数必须在给定定义域上一一对应；第一个恒等式中 \(x\) 属于原函数定义域，第二个恒等式中 \(y\) 属于原函数值域。
  \item \textbf{典型题型：} 反函数求值、复合函数化简、由图像或定义域值域判断反函数。
  \item \textbf{易错点：} \(f^{-1}(x)\) 是反函数，不是 \(\frac{1}{f(x)}\)；若忽略变量范围，可能把无定义的复合式误写成恒等式。
  \item \textbf{关联知识点：} \knowledgeRef[inverse-function]{反函数}（第 1 讲“反函数的定义、存在条件与图像关系”）。
\end{itemize}

\textbf{倒数代换中的根式化简}

当 \(x>0\) 时，
\[
\sqrt{1+\frac{1}{x^2}}
=\sqrt{\frac{x^2+1}{x^2}}
=\frac{\sqrt{1+x^2}}{x}.
\]

\begin{itemize}
  \item \textbf{使用条件：} 必须明确 \(x>0\)。一般情况下 \(\sqrt{x^2}=|x|\)，不能直接写成 \(x\)。
  \item \textbf{典型题型：} 将关系式中的 \(x\) 换为 \(\frac{1}{x}\) 后化简右端。
  \item \textbf{易错点：} 忘记绝对值导致符号错误。
  \item \textbf{关联题目：} \problemRef{MATH1-CALC-0002}。
\end{itemize}

\textbf{反双曲正弦与双曲正弦互为反函数}

\[
\ln\left(x+\sqrt{x^2+1}\right)=y
\quad\Longleftrightarrow\quad
x=\frac12\left(e^y-e^{-y}\right).
\]

等价地，
\[
\operatorname{arsinh}x=\ln\left(x+\sqrt{x^2+1}\right),
\qquad
\sinh x=\frac{e^x-e^{-x}}{2}.
\]

\begin{itemize}
  \item \textbf{使用条件：} \(x+\sqrt{x^2+1}>0\)，原函数在 \(\mathbb{R}\) 上有定义并一一对应。
  \item \textbf{典型题型：} 求 \(\ln\left(x+\sqrt{x^2+1}\right)\) 的反函数，或识别反双曲正弦、双曲正弦结构。
  \item \textbf{易错点：} 直接平方硬解容易变繁；倒数有理化可直接得到 \(e^{-y}=\sqrt{x^2+1}-x\)。
  \item \textbf{关联题目：} \problemRef{MATH1-CALC-0003}。
\end{itemize}

\textbf{复合函数定义域条件}

若 \(f\) 的定义域为 \(D_f\)，\(g\) 的定义域为 \(D_g\)，则
\[
D_{f\circ g}=\{x\in D_g\mid g(x)\in D_f\}.
\]

\begin{itemize}
  \item \textbf{使用条件：} 先保证内层 \(g(x)\) 有意义，再保证内层输出能作为外层 \(f\) 的输入。
  \item \textbf{典型题型：} 复合函数定义域、分段函数复合、已知复合结果反求中间函数。
  \item \textbf{易错点：} 把复合函数定义域误认为内层函数定义域，漏掉外层函数对内层输出的限制。
  \item \textbf{关联题目：} \problemRef{MATH1-CALC-0004}，\problemRef{MATH1-CALC-0005}。
\end{itemize}

\textbf{平方复合反求中间函数}

若 \(f(t)=t^2\)，且
\[
f[\varphi(x)]=G(x),
\]
则
\[
[\varphi(x)]^2=G(x).
\]
若进一步已知 \(\varphi(x)\ge 0\)，则
\[
\varphi(x)=\sqrt{G(x)}.
\]

\begin{itemize}
  \item \textbf{使用条件：} \(G(x)\ge 0\)，并且必须有分支条件才能唯一确定 \(\varphi(x)\)。
  \item \textbf{典型题型：} 已知平方型外层函数和复合结果，反求内层函数。
  \item \textbf{易错点：} 没有 \(\varphi(x)\ge 0\) 等条件时，不能只取正根。
  \item \textbf{关联题目：} \problemRef{MATH1-CALC-0004}。
\end{itemize}

\textbf{函数有界性的定义}

函数 \(f(x)\) 在区间 \(I\) 上有界，等价于存在正数 \(M\)，使得
\[
|f(x)|\le M,\qquad x\in I.
\]

\begin{itemize}
  \item \textbf{使用条件：} 必须先明确讨论区间 \(I\)。
  \item \textbf{典型题型：} 证明函数有界，或判断在不同区间上的有界性。
  \item \textbf{易错点：} 只证明上有界而忘记下有界；未取绝对值。
  \item \textbf{关联题目：} \problemRef{MATH1-CALC-0006}。
\end{itemize}

\textbf{基本不等式型放缩}

\[
1+x^2\ge 2|x|,\qquad x\in\mathbb{R}.
\]

\begin{itemize}
  \item \textbf{使用条件：} 由 \(\frac{1+x^2}{2}\ge |x|\) 得到。
  \item \textbf{典型题型：} 处理 \(\frac{|x|}{1+x^2}\) 一类有界性证明。
  \item \textbf{易错点：} 当需要除以 \(|x|\) 时必须单独讨论 \(x=0\)。
  \item \textbf{关联题目：} \problemRef{MATH1-CALC-0006}。
\end{itemize}

\textbf{常见放缩公式清单}

\[
a^2+b^2\ge 2ab,\qquad a,b\in\mathbb{R}.
\]

\[
a+\frac1a\ge 2,\qquad a>0.
\]

\[
\frac{a+b}{2}\ge \sqrt{ab},\qquad a\ge0,\ b\ge0.
\]

\[
|a+b|\le |a|+|b|,\qquad
\bigl||a|-|b|\bigr|\le |a-b|.
\]

\[
\sqrt{x^2+a}\ge |x|,\qquad a\ge0.
\]

若 \(A,B,C>0\)，且 \(A\ge B\)，则
\[
\frac{C}{A}\le \frac{C}{B}.
\]

\begin{itemize}
  \item \textbf{使用条件：} 放缩时必须确认正负号，尤其是分母放缩只在正分母下可直接保持方向。
  \item \textbf{典型题型：} 有界性、估值、极限夹逼、证明不等式、避免繁琐求导求最值。
  \item \textbf{易错点：} 放大分母会使正分式变小；若分子或分母可能变号，必须先取绝对值或分情况。
  \item \textbf{关联题目：} \problemRef{MATH1-CALC-0006}。
\end{itemize}

\textbf{单调性的乘积判别式}

对任意 \(x_1,x_2\in I\)，\(x_1\ne x_2\)，有
\[
f\text{ 严格递增}
\quad\Longleftrightarrow\quad
(x_1-x_2)[f(x_1)-f(x_2)]>0,
\]
\[
f\text{ 严格递减}
\quad\Longleftrightarrow\quad
(x_1-x_2)[f(x_1)-f(x_2)]<0.
\]

\begin{itemize}
  \item \textbf{使用条件：} 在同一区间 \(I\) 内任取两点。
  \item \textbf{典型题型：} 无导数条件下判断或证明单调性。
  \item \textbf{易错点：} 忘记 \(x_1<x_2\) 时 \(x_1-x_2<0\)，导致符号判断反了。
  \item \textbf{关联题目：} \problemRef{MATH1-CALC-0007}。
\end{itemize}

\textbf{奇偶函数判定与分解}

\[
f(-x)=f(x)\Longleftrightarrow f\text{ 为偶函数},
\qquad
f(-x)=-f(x)\Longleftrightarrow f\text{ 为奇函数}.
\]

在定义域关于原点对称时，
\[
\frac12[f(x)+f(-x)]\text{ 为偶函数},\qquad
\frac12[f(x)-f(-x)]\text{ 为奇函数}.
\]

\begin{itemize}
  \item \textbf{使用条件：} 定义域必须关于原点对称。
  \item \textbf{典型题型：} 奇偶性证明、图像对称、函数方程赋值。
  \item \textbf{易错点：} 没有先检查定义域；混淆奇函数和偶函数条件。
  \item \textbf{关联题目：} \problemRef{MATH1-CALC-0008}。
\end{itemize}

\textbf{周期函数的基本判定}

\[
f(x+T)=f(x)\quad (T\ne0)
\]
表示 \(T\) 是 \(f(x)\) 的一个周期。

\begin{itemize}
  \item \textbf{使用条件：} 对定义域内所有相关 \(x\) 成立，且 \(x+T\) 仍在定义域内。
  \item \textbf{典型题型：} 由递推关系、三角诱导公式或复合结构证明周期性。
  \item \textbf{易错点：} 证明一个周期不等于证明最小正周期。
  \item \textbf{关联题目：} \problemRef{MATH1-CALC-0009}。
\end{itemize}
\end{knowledgeBox}
